    No.1 Menghitung Rata-rata dari 5 nilai¶
    a=int(input('Masukkan Nilai A = '))
b=int(input('Masukkan Nilai B = '))
c=int(input('Masukkan Nilai C = '))
d=int(input('Masukkan Nilai D = '))
e=int(input('Masukkan Nilai E = '))
mean=(a+b+c+d+e)/5
print('Rata-rata =',mean)

    Masukkan Nilai A = 1
Masukkan Nilai B = 2
Masukkan Nilai C = 3
Masukkan Nilai D = 4
Masukkan Nilai E = 5
Rata-rata = 3.0
    No.2 Menghitung Luas & Keliling Persegi Panjang¶
    panjang=int(input('Masukkan Panjang = '))
lebar=int(input('Masukkan Lebar = '))
luas=panjang*lebar
keliling=2*(panjang+lebar)
print('Luas Persegi Panjang =',luas)
print('Keliling Persegi Panjang=',keliling)

    Masukkan Panjang = 25
Masukkan Lebar = 10
Luas Persegi Panjang = 250
Keliling Persegi Panjang = 70
    No.3 Menghitung BMI¶
    berat=int(input('Masukkan Berat = '))
tinggi=float(input('Masukkan Tinggi = '))
bmi=berat/(tinggi*tinggi)
print('Nilai BMI =',bmi)
if(bmi<18.5):
    print('Berat Badan Kurang')
elif(bmi>=18.5 and bmi<=22.9):
    print('Berat Badan Normal')    
elif(bmi>=23 and bmi<=29.9):
    print('Berat Badan Berlebih (Kecenderungan Obesitas)')
elif(bmi>30):
    print('Obesitas')   

    Masukkan Berat = 75
Masukkan Tinggi = 1.65
Nilai BMI = 27.548209366391188
Berat Badan Berlebih (kecenderungan obesitas)
    No.4 Menentukan Nilai Minimal dan Maksimal¶
    n=int(input('Masukkan Banyaknya Bilangan ='))
bil=[]
for x in range (1,n+1):
    a=int(input('Bilangan ke-'+str(x)+' = '))    
    bil.append(a)

print('Bilangan Terkecil = ',min(bil))
print('Bilangan Terbesar = ',max(bil))

    Masukkan Banyaknya Nilai =3
Bilangan ke-1 = 2
Bilangan ke-2 = 3
Bilangan ke-3 = 45
Bilangan Terkecil =  2
Bilangan Terbesar =  45
